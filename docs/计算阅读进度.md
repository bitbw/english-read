在 EPUB.js 中，可以通过以下方式计算全文阅读进度：

核心思路

通过 rendition.location 获取当前位置，再与总位置对比计算百分比。

实现方法

1. 获取当前位置和总位置

// 初始化 epub
const book = ePub(url);
const rendition = book.renderTo("viewer", {
  width: "100%",
  height: "100%"
});

// 获取阅读进度
function getReadingProgress() {
  if (!rendition || !rendition.location) {
    return 0;
  }
  
  const currentLocation = rendition.location.atStart;
  const totalLocations = rendition.location.total;
  
  if (totalLocations <= 0) return 0;
  
  // 计算进度百分比
  const progress = (currentLocation / totalLocations) * 100;
  return Math.min(progress, 100); // 确保不超过100%
}


2. 监听翻页事件更新进度

// 监听位置变化
rendition.on("relocated", function(location) {
  const progress = getReadingProgress();
  console.log(`阅读进度: ${progress.toFixed(2)}%`);
  
  // 更新UI或存储进度
  updateProgressUI(progress);
});

// 翻页时也更新
rendition.on("rendered", function() {
  const progress = getReadingProgress();
  updateProgress(progress);
});


3. 完整的进度管理类

class ReadingProgress {
  constructor(rendition) {
    this.rendition = rendition;
    this.progress = 0;
    this.init();
  }
  
  init() {
    // 监听位置变化
    this.rendition.on("relocated", (location) => {
      this.updateProgress();
    });
    
    // 监听章节变化
    this.rendition.on("rendered", (section) => {
      this.updateProgress();
    });
  }
  
  updateProgress() {
    if (!this.rendition.location) return;
    
    const { atStart, total } = this.rendition.location;
    
    if (total > 0) {
      this.progress = (atStart / total) * 100;
      this.progress = Math.min(this.progress, 100);
      
      // 触发自定义事件
      this.onProgressChange(this.progress);
    }
  }
  
  // 跳转到指定进度
  gotoProgress(percent) {
    if (!this.rendition || !this.rendition.location) return;
    
    const total = this.rendition.location.total;
    const targetLocation = Math.floor((percent / 100) * total);
    
    this.rendition.display(targetLocation);
  }
  
  onProgressChange(progress) {
    console.log(`进度更新: ${progress.toFixed(2)}%`);
    // 这里可以更新UI、保存到本地存储等
  }
  
  getCurrentProgress() {
    return this.progress;
  }
}

// 使用示例
const progressManager = new ReadingProgress(rendition);


4. 保存和恢复阅读进度

// 保存进度到本地存储
function saveProgress(progress) {
  localStorage.setItem('bookProgress', progress);
  localStorage.setItem('lastReadTime', new Date().toISOString());
}

// 恢复进度
function restoreProgress() {
  const savedProgress = localStorage.getItem('bookProgress');
  if (savedProgress) {
    progressManager.gotoProgress(parseFloat(savedProgress));
  }
}


5. 基于章节的进度计算（更精确）

// 如果需要基于章节数计算进度
async function getChapterBasedProgress() {
  const spine = book.spine;
  const totalChapters = spine.items.length;
  
  // 获取当前章节索引
  const currentLocation = rendition.location;
  const currentChapterIndex = currentLocation.start.index;
  
  // 计算每个章节的平均位置长度
  const locationsPerChapter = rendition.location.total / totalChapters;
  
  // 计算进度
  const chapterProgress = (currentChapterIndex / totalChapters) * 100;
  const inChapterProgress = (currentLocation.atStart % locationsPerChapter) / locationsPerChapter * (100 / totalChapters);
  
  return chapterProgress + inChapterProgress;
}


注意事项

1. 位置计算的准确性：rendition.location.total 可能不会立即可用，需要等待书籍完全加载
2. 事件触发频率：relocated 事件在每次位置变化时触发，可能比较频繁
3. 性能考虑：更新进度时避免频繁的DOM操作
4. 跨会话保存：建议将进度保存到 localStorage 或服务器

简化版本

// 最简单的实现
rendition.on("relocated", function() {
  const location = rendition.location;
  if (location && location.total > 0) {
    const progress = (location.atStart / location.total) * 100;
    console.log(`进度: ${progress.toFixed(1)}%`);
  }
});


这样就能实时计算并跟踪用户的阅读进度了。
